## 2.증감연산자

```java
package Java0511;

public class ex02_증감연산자 {
	
	public static void main(String[] args) {
		// 단항연산자
		// 증가 감소 연산자( ++ ,-- )
		
		// ++num : num이 가지고 있는 값에 1을 증하간 뒤에 연산에 적용
		// num++ : num이 가지고 있는 값을 연산에 적용한 뒤 1을 증가
		// 연산에 적용한 뒤 1을 증가
		
		int num = 5;
		System.out.println("num값 : " + num);
		
		++num; //바로 1이 증가할 수 있음
		
		System.out.println("++num값 : "+ num );
		
		num++;
		
		System.out.println("num++값 :" + num);
		
		int num1 =5;
		int result1;
		
		result1 = ++num1;
		//result1 = ++num1;
		//num1 = num1 +1;
		//result1 =num1;
		
		System.out.println("++num1 result1 :" + result1);
		
		result1 = num1++;
		//result1 = num1++;
		//result1 = num1;
		//num1 = num1 + 1;
		
		
		System.out.println("num1++ result1 :" + result1);
		
		int num2 = 10;
		--num2; //9
		
		--num2; //8
		num2--;
		System.out.println(num2);
		
		//num2 = num2 -1;
		//num2 = 10 -1
		//num2 = 9 -1 
		//result2 =num2;
		//result2 =8;
		
		int result2;
		result2 = num2--;
		System.out.println("num2 : " + result2);
		//result2 = num2--
		//result2 = num2
		//result2 =8
		//num2 = num2 -1
		//num2 = 8 -1
		
		int result3;
		System.out.println("*num1: "+num1 + " *num2: " +num2);
		//		=   7         6
		result3 = ++num1 + ++num2;
		//result3=  8         7
		//8+7   -> 15
		System.out.println(" ");
		System.out.println("(1)num1 : " + num1);    
		System.out.println("(1)num2 : " + num2);	
		System.out.println("(1)result : " + result3);
		//      =   8        7
		result3 =++num1 + num2++;
		//result3=  9        7
		//9+7    -> 16
		//num1=9 ,num2=8
		
		System.out.println(" ");
		System.out.println("(2)num1 : " + num1);
		System.out.println("(2)num2 : " + num2);
		System.out.println("(2)result : " + result3);
		//      =   9        8
		result3 =num1++ + ++num2;
		//result=   9        9
		//9+9    -> 18
		//num1=10,num2=9
		System.out.println(" ");
		System.out.println("(3)num1 : " + num1);
		System.out.println("(3)num2 : " + num2);
		System.out.println("(3)result : " + result3);
		//        10        9
		result3 =num1-- + --num2;
		//result3= 10       8
		//10+8   -> 18
		//num1=9,num2=8
		System.out.println(" ");
		System.out.println("(3)num1 : " + num1);
		System.out.println("(3)num2 : " + num2);
		System.out.println("(3)result : " + result3);	
	
		int num3 = 5;
		int num4 = 11;
		int result4;
		
		result4 = ++num3 + --num4; //16 , 6,10
		System.out.println(result4);  //16
		result4 = num4++ - num3--; //4  , 5,11
		System.out.println(result4);  //4
		result4 = num3++ - ++num4; //-7 , 6,12
		System.out.println(result4);  //-7
	}
}
```

## 3.대입연산자
```java
/*
 Data : 2020.05.11
 Author : 안준필
 Description : 대입연산자(assignmentOperation)
 Version : 1.0  
 */


package Java0511;

public class ex03_대입연산자 {
	public static void main(String[] args) {
		int age = 35;
		// 나이를 의미하는 변수 age에 변수값 35를 대입한다
		// 왼쪽변수 = 오른쪽변수(또는 상수 ,식)
		// 1value(항상 변수) = rvalue(상수,변수,식)
		
		
		int num = 1;
		age = num;
		age = 1+25;
		
		
		// 부호연산자 : (+),(-)
		int num1 = 10;
		
		System.out.println(+num1); //10
		System.out.println(-num1); //-10 
		System.out.println(num1);  //10
		
		num1= -num1;
				
		System.out.println(num1); //-10
	}
}
```
## 4.산술연산자

```java
/*
 Data : 2020.05.11
 Author : 안준필
 Description : 산술연산자(Operation)
 Version : 1.0  
 */

package Java0511;

public class ex04_산술연산자 {
	public static void main(String[] args) {
		
		//정수형 변수 2개선언
		//+ , - ,* , / ,% 연산수행
		int num1 = 10,num2 = 5,result;
		
		
		//덧샘
		result = num1 + num2;
		System.out.println("덧셈 : "+result);
		
		//뺄셈
		result = num1 - num2;
		System.out.println("뺄셈 : "+result);
		
		//곱셈
		result = num1 * num2;
		System.out.println("곱셈 : "+result);
		
		//나눗셈(몫)
		result = num1 / num2;
		System.out.println("나눗셈(몫) : "+result);
		
		//나머지
		result = num1 % num2;
		System.out.println("나머지 : "+result);
	}
}
```

## 5.관계연산자
```java
/*
 Data : 2020.05.11
 Author : 안준필
 Description : 관계연산자(Operation)
 Version : 1.0  
 */

package Java0511;

public class EX05_관계연산자 {
	public static void main(String[] args) {
		//정수형 변수 2개 선언
		
		int num1=10, num2=5;
		
		// >, < , >= , <= ,== ,!= 연산 수행
		
		boolean result; //result 변수를 boolean 으로 선언
		
		result = num1 > num2;
		System.out.println("num1 > num2 : "+ result);
		
		result = num1 < num2;
		System.out.println("num1 < num2 : "+ result);
		
		result = num1 == num2;
		System.out.println("num1 == num2 : "+ result);
		
		result = num1 >= num2;
		System.out.println("num1 >= num2 : "+ result);
		
		result = num1 <= num2;
		System.out.println("num1 <= num2 : "+ result);
		
		result = num1 != num2;
		System.out.println("num1 != num2 : "+ result);
	}
}
```
