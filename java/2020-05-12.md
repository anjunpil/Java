```java
		max=Math.max(max,num3);
```

```java
package java0512;

import java.util.Scanner;

public class ex04_switchCase {
	public static void main(String[] args) {
		
		/*
		   switch(조건변수){
		   	 	case 변수값1:
		   	 		해당조건이 맞을 때 실행
		   	 		break;
		   	 	case 변수값2:
		   	 		해당조건이 맞을 때 실행
		   	 		break;
		   	 	case 변수값3:
		   	 		해당조건이 맞을 때 실행
		   	 		break;	
		   		
		   		default:
		   			case에서 어떤 조건도 맞지 않을 때 실행
		   			break;
		   }  
		 */
		
		Scanner sc = new Scanner(System.in);
		
		System.out.print("숫자입력 : ");
		int num =sc.nextInt() ;
		
		switch(num) {
		case 0:
			System.out.println("0을 입력");
			break;
		case 1:
			System.out.println("1을 입력");
			break;
		case 2:
			System.out.println("2을 입력");
			break;
		default :
			System.out.println("그 외	 숫자입력");
			break;
		}
		
		//switch/case문은 if else 구조로 변경이 가능하지만 
		//if else 구조로 작성된 모든 코드를 switch 문으로 변경할 수는 없다.
		//break 갑 없으면 연속으로 실행
	}
}
```

```java
package java0512;

public class ex07_loopfor {
	public static void main(String[] args) {
		//안녕하세요 10번 출력
		// 
		/*
		 for(초기화식 ; 조건식 ; 중감식){
		 	반복할 수행문
		 }
		 */
		
		int i;
		
		for (i=0;i<10;i+=2) {
			System.out.println(i+ " 안녕하세요");
		}
		
		System.out.println(i);
		
		//1부터 10까지의 합을 출력
		int Sum=0;
		for (i = 1;i<=10;i++) {
			Sum+=i;
			System.out.println("i : " +i + "   합계 : " +Sum );
		}
		System.out.println(Sum);
		
		
	}
}
```
